# 问题根因
网络原因，导致安装超时

# 解决方案
执行用例之前配置一个可以访问的pip源，如下所示：
mkdir ~/.pip
echo "[global]
index-url = https://mirrors.aliyun.com/pypi/simple" > ~/.pip/pip.conf
